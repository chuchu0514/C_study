    int b = 20;

}

main

/삼중 포인터를 왜 쓰냐 이중포인터의 값을 변경하기 위해서인데.. 이걸 함수로 변경을 하려면 이제 값을 복사하는 게 아니고 주소를 불러와야 하므로
이중포인터의 주소를 불러올 수 있는 삼중포인터를 쓰는 것 
void changeDoublePointer(int ***ppp) { ppp ──► pp ──► p ──► a  (ppp는 삼중포인터가 맞다 저기서 )
    static int b = 20;               int***   int**  int*   int 
    static int *newp = &b;   // 새로운 포인터
    static int **newpp = &newp; // 새로운 이중포인터

    *ppp = newpp;   // pp 자체를 newpp로 변경 //*ppp의 의미는 pp겠지 타입은 int**그래서 저 위의 타입을 맞춰준것 newpp도 별을 안 붙혔으니 타입 int**
    즉! *ppp = &newp; 와 같은 말이다 
}

int main() {
    int a = 10;

    int *p = &a;
    int **pp = &p; //참고로 선언을 할 땐 int **형인거임 선언하고 나서 **pp 쓰면 그냥 int형 pp이렇게 써야 **int형임 pp의 타입 = int **이라고 선언순간에 컴파일러가 처리함 

    printf("before: %d\n", **pp);

    changeDoublePointer(&pp);  // 삼중포인터 전달

    printf("after: %d\n", **pp);

    return 0;
}

pp p의 주소 즉 pp = &p
*pp pp가 가리키는 값 즉 *pp = p , 출력은 a의 주소
**pp는   pp → p → a 즉 a의 값
